import adsk.core, adsk.fusion, traceback

def run(context):
    ui = None
    try:
        app = adsk.core.Application.get()
        ui  = app.userInterface

        
        # Get active design
        product = app.activeProduct
        design = adsk.fusion.Design.cast(product)
 
        # Get root component in this design
        rootComp = design.rootComponent
        
        # Get OffsetFaces feature
        offsetFaces = rootComp.features.offsetFacesFeatures
        
        for feature in offsetFaces:
            featureInfo = 'FeatureInfo: name - {}, isSuppressed - {}, isParametric - {}, bodies - {}'.format(
            feature.name, feature.isSuppressed, feature.isParametric, feature.bodies.count)
            ui.messageBox(featureInfo)
        
        # Get the first sub component
        subComp = rootComp.occurrences.item(0).component
        
        # Get OffsetFaces feature by name
        offsetFacesFeat = subComp.features.offsetFacesFeatures.itemByName('OffsetFaces1')
        featureInfo = 'FeatureInfo: name - {}, isSuppressed - {}, isParametric - {}, bodies - {}'.format(
            offsetFacesFeat.name, offsetFacesFeat.isSuppressed, offsetFacesFeat.isParametric, offsetFacesFeat.bodies.count)
        ui.messageBox(featureInfo)
        

    except:
        if ui:
            ui.messageBox('Failed:\n{}'.format(traceback.format_exc()))
#include <Core/Application/Application.h>
#include <Core/UserInterface/UserInterface.h>
#include <Core/Application/Document.h>

#include <Fusion/Fusion/Design.h>
#include <Fusion/Components/Component.h>
#include <Fusion/Components/Occurrences.h>
#include <Fusion/Components/Occurrence.h>
#include <Fusion/Features/Features.h>
#include <Fusion/Features/OffsetFacesFeatures.h>
#include <Fusion/Features/OffsetFacesFeature.h>
#include <Fusion/BRep/BRepBodies.h>

#include <iostream>



using namespace adsk::core;
using namespace adsk::fusion;

Ptr<UserInterface> ui;

extern "C" XI_EXPORT bool run(const char* context)
{
	Ptr<Application> app = Application::get();
	if (!app)
		return false;

	ui = app->userInterface();
	if (!ui)
		return false;


	// Get active design
	Ptr<Product> product = app->activeProduct();
	if (!product)
		return false;

	Ptr<Design> design = product;
	if (!design)
		return false;

	// Get root component in this design
	Ptr<Component> rootComp = design->rootComponent();
	if (!rootComp)
		return false;

	// Get OffsetFaces feature
	Ptr<Features> features = rootComp->features();
	if (!features)
		return false;

	Ptr<OffsetFacesFeatures> offsetFaces = features->offsetFacesFeatures();
	if (!offsetFaces)
		return false;

	for (size_t i = 0; i < offsetFaces->count(); ++i)
	{
		Ptr<OffsetFacesFeature> feature = offsetFaces->item(i);
		if (!feature)
			return false;

		Ptr<BRepBodies> bodies = feature->bodies();
		if (!bodies)
			return false;

		std::string featureInfo = "FeatureInfo: name - " + feature->name() + ", isSuppressed - " + std::to_string(feature->isSuppressed()) +
			", isParametric - " + std::to_string(feature->isParametric()) + ", bodies - " + std::to_string(bodies->count());
		ui->messageBox(featureInfo);

	}

	// Get the first sub component
	Ptr<Occurrences> subOccs = rootComp->occurrences();
	if (!subOccs)
		return false;

	Ptr<Occurrence> firstOcc = subOccs->item(0);
	if (!firstOcc)
		return false;
		
	Ptr<Component> subComp = firstOcc->component();
	if (!subComp)
		return false;

	// Get OffsetFaces feature by name
	Ptr<Features> subFeatures = subComp->features();
	if (!subFeatures)
		return false;

	Ptr<OffsetFacesFeatures> subOffsetFaces = subFeatures->offsetFacesFeatures();
	if (!subOffsetFaces)
		return false;

	Ptr<OffsetFacesFeature> subOffsetFacesFeature = offsetFaces->itemByName("OffsetFaces1");
	if (!subOffsetFacesFeature)
		return false;

	Ptr<BRepBodies> subBodies = subOffsetFacesFeature->bodies();
	if (!subBodies)
		return false;

	std::string featureInfo = "FeatureInfo: name - " + subOffsetFacesFeature->name() + ", isSuppressed - " + std::to_string(subOffsetFacesFeature->isSuppressed()) +
		", isParametric - " + std::to_string(subOffsetFacesFeature->isParametric()) + ", bodies - " + std::to_string(subBodies->count());

	ui->messageBox(featureInfo);


	return true;
}
